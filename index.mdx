## Quick Start

Get up and running with Slick in minutes.

<Steps>
  <Step title="Install via CDN or package manager">
    ```html
    <link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/slick-carousel@1.8.1/slick/slick.css"/>
    <script src="https://cdn.jsdelivr.net/npm/slick-carousel@1.8.1/slick/slick.min.js"></script>
    ```
  </Step>
  
  <Step title="Add your HTML markup">
    ```html
    <div class="your-slider">
      <div><img src="image1.jpg" alt="Slide 1"></div>
      <div><img src="image2.jpg" alt="Slide 2"></div>
      <div><img src="image3.jpg" alt="Slide 3"></div>
    </div>
    ```
  </Step>
  
  <Step title="Initialize Slick">
    ```javascript
    $('.your-slider').slick({
      dots: true,
      infinite: true,
      speed: 500,
      slidesToShow: 1,
      slidesToScroll: 1
    });
    ```
  </Step>
</Steps>
